# Guía de comandos usados - Actividad 2

git checkout -b nueva-rama
// Crear y cambiar a una nueva rama

git status
// Ver el estado del repositorio

git add .
git commit -m "Cambios en la nueva rama"
// Guardar cambios

git checkout main
// Volver a la rama principal

git merge nueva-rama
// Fusionar cambios de otra rama

git pull origin main
// Traer cambios del repositorio remoto

git push origin main
// Subir cambios a GitHub

// Colaboración
// colaboradoren en GitHub
// Clonar repo: git clone https://github.com/RicharRamos/Actividad2-Git.git
// Pull request: Desde GitHub, comparar y unir ramas
